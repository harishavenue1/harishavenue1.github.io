Playwright Automation Reference Guide

Basic Setup

Browser Launch

// JavaScript/TypeScript
const { chromium, firefox, webkit } = require('playwright');
const browser = await chromium.launch();
const page = await browser.newPage();

// Python
from playwright.sync_api import sync_playwright
with sync_playwright() as p:
    browser = p.chromium.launch()
    page = browser.new_page()

// C#
using Microsoft.Playwright;
var playwright = await Playwright.CreateAsync();
var browser = await playwright.Chromium.LaunchAsync();
var page = await browser.NewPageAsync();

Navigation

// JavaScript/TypeScript
await page.goto('https://example.com');
await page.goBack();
await page.goForward();
await page.reload();

// Python
page.goto('https://example.com')
page.go_back()
page.go_forward()
page.reload()

// C#
await page.GotoAsync("https://example.com");
await page.GoBackAsync();
await page.GoForwardAsync();
await page.ReloadAsync();

Element Selection

// JavaScript/TypeScript
page.locator('selector')                    // CSS selector
page.locator('#id')                        // By ID
page.locator('.class')                     // By class
page.locator('text=Button Text')           // By text
page.locator('input[placeholder="Email"]') // By attribute
page.getByRole('button', { name: 'Submit' }) // By role
page.getByText('Click me')                 // By text
page.getByLabel('Email')                   // By label
page.getByPlaceholder('Enter email')       // By placeholder

// Python
page.locator('selector')
page.get_by_role('button', name='Submit')
page.get_by_text('Click me')
page.get_by_label('Email')

// C#
page.Locator("selector");
page.GetByRole(AriaRole.Button, new() { Name = "Submit" });
page.GetByText("Click me");
page.GetByLabel("Email");

User Actions

Clicking & Typing

// JavaScript/TypeScript
await page.locator('button').click();
await page.locator('input').fill('text');
await page.locator('input').clear();
await page.locator('input').type('text');
await page.locator('button').dblclick();
await page.locator('element').hover();

// Python
page.locator('button').click()
page.locator('input').fill('text')
page.locator('input').clear()
page.locator('input').type('text')

// C#
await page.Locator("button").ClickAsync();
await page.Locator("input").FillAsync("text");
await page.Locator("input").ClearAsync();

Form Controls

// JavaScript/TypeScript
await page.locator('select').selectOption('value');
await page.locator('checkbox').check();
await page.locator('checkbox').uncheck();
await page.locator('input[type="file"]').setInputFiles('path/to/file');

// Python
page.locator('select').select_option('value')
page.locator('checkbox').check()
page.locator('checkbox').uncheck()
page.locator('input[type="file"]').set_input_files('path/to/file')

// C#
await page.Locator("select").SelectOptionAsync("value");
await page.Locator("checkbox").CheckAsync();
await page.Locator("checkbox").UncheckAsync();

Assertions

// JavaScript/TypeScript
await expect(page.locator('element')).toBeVisible();
await expect(page.locator('element')).toBeHidden();
await expect(page.locator('element')).toHaveText('text');
await expect(page.locator('element')).toContainText('text');
await expect(page.locator('input')).toHaveValue('value');
await expect(page.locator('element')).toHaveAttribute('attr', 'value');
await expect(page).toHaveTitle('Page Title');
await expect(page).toHaveURL('https://example.com');

// Python
expect(page.locator('element')).to_be_visible()
expect(page.locator('element')).to_have_text('text')
expect(page.locator('input')).to_have_value('value')
expect(page).to_have_title('Page Title')

// C#
await Expect(page.Locator("element")).ToBeVisibleAsync();
await Expect(page.Locator("element")).ToHaveTextAsync("text");
await Expect(page.Locator("input")).ToHaveValueAsync("value");

Waits

// JavaScript/TypeScript
await page.waitForSelector('selector');
await page.waitForLoadState('networkidle');
await page.waitForTimeout(5000);
await page.waitForURL('**/dashboard');
await page.locator('element').waitFor();

// Python
page.wait_for_selector('selector')
page.wait_for_load_state('networkidle')
page.wait_for_timeout(5000)
page.locator('element').wait_for()

// C#
await page.WaitForSelectorAsync("selector");
await page.WaitForLoadStateAsync(LoadState.NetworkIdle);
await page.WaitForTimeoutAsync(5000);

Screenshots & Videos

// JavaScript/TypeScript
await page.screenshot({ path: 'screenshot.png' });
await page.locator('element').screenshot({ path: 'element.png' });

// Python
page.screenshot(path='screenshot.png')
page.locator('element').screenshot(path='element.png')

// C#
await page.ScreenshotAsync(new() { Path = "screenshot.png" });
await page.Locator("element").ScreenshotAsync(new() { Path = "element.png" });

Keyboard & Mouse Actions

// JavaScript/TypeScript
await page.keyboard.press('Enter');
await page.keyboard.type('Hello World');
await page.locator('element').click({ button: 'right' }); // Right click
await page.locator('source').dragTo(page.locator('target')); // Drag & drop

// Python
page.keyboard.press('Enter')
page.keyboard.type('Hello World')
page.locator('element').click(button='right')
page.locator('source').drag_to(page.locator('target'))

// C#
await page.Keyboard.PressAsync("Enter");
await page.Keyboard.TypeAsync("Hello World");
await page.Locator("element").ClickAsync(new() { Button = MouseButton.Right });

Frames & Network Interception

// JavaScript/TypeScript
const frame = page.frame('frameName');
await frame.locator('input').fill('text');

// Network interception
await page.route('**/api/users', route => {
    route.fulfill({ json: { users: [] } });
});
await page.route('**/api/**', route => route.continue());

// Python
frame = page.frame('frameName')
frame.locator('input').fill('text')

# Network interception
page.route('**/api/users', lambda route: route.fulfill(json={"users": []}))

// C#
var frame = page.Frame("frameName");
await frame.Locator("input").FillAsync("text");

Cookies & Mobile Emulation

// JavaScript/TypeScript
await context.addCookies([{ name: 'token', value: 'abc123', url: 'https://example.com' }]);
const cookies = await context.cookies();
await context.clearCookies();

// Mobile emulation
const iPhone = playwright.devices['iPhone 12'];
const context = await browser.newContext({ ...iPhone });

// Python
context.add_cookies([{"name": "token", "value": "abc123", "url": "https://example.com"}])
cookies = context.cookies()
context.clear_cookies()

# Mobile emulation
iphone = playwright.devices['iPhone 12']
context = browser.new_context(**iphone)

// C#
await context.AddCookiesAsync(new[] { new Cookie { Name = "token", Value = "abc123", Url = "https://example.com" } });
var cookies = await context.CookiesAsync();
await context.ClearCookiesAsync();

Multi-Browser & Context

// JavaScript/TypeScript
const context = await browser.newContext();
const page1 = await context.newPage();
const page2 = await context.newPage();
await context.close();

// Python
context = browser.new_context()
page1 = context.new_page()
page2 = context.new_page()
context.close()

// C#
var context = await browser.NewContextAsync();
var page1 = await context.NewPageAsync();
var page2 = await context.NewPageAsync();
await context.CloseAsync();
