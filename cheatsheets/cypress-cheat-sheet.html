Cypress Commands Reference Guide

Basic Commands

Navigation

cy.visit('https://example.com')           // Navigate to URL
cy.go('back')                            // Browser back
cy.go('forward')                         // Browser forward
cy.go(-1)                               // Go back 1 page
cy.go(1)                                // Go forward 1 page
cy.reload()                             // Reload page

Element Selection

cy.get('selector')                       // Get element by CSS selector
cy.get('input[placeholder="Email"]')     // Get by attribute
cy.get('#id')                           // Get by ID
cy.get('.class')                        // Get by class
cy.contains('text')                     // Get element containing text
cy.contains('Sign in').click()          // Chain with action

XPath (requires cypress-xpath plugin)

cy.xpath('//a[normalize-space()="Sign in"]')
cy.xpath('//input[@placeholder="Email"]')
cy.xpath('//button[normalize-space()="Sign in"]')

User Actions

Clicking

cy.get('button').click()                 // Basic click
cy.get('element').click({force: true})   // Force click (bypass visibility)

Typing

cy.get('input').type('text')             // Type text
cy.get('input').clear()                  // Clear input
cy.get('input').clear().type('new text') // Clear and type

Form Controls

// Checkboxes
cy.get('#checkbox').check()              // Check checkbox
cy.get('#checkbox').check({force: true}) // Force check
cy.get('#checkbox').uncheck()            // Uncheck

// Radio Buttons
cy.get('#radio').click({force: true})    // Select radio button

// Dropdowns
cy.get('select').select('option')        // Select by text
cy.get('select').select('value')         // Select by value
cy.get('#mySelect').select('Set to 100%').should('have.value', '100%')

Assertions & Validations

Visibility & State

cy.get('element').should('be.visible')
cy.get('element').should('not.be.visible')
cy.get('element').should('exist')
cy.get('element').should('not.exist')
cy.get('checkbox').should('be.checked')
cy.get('checkbox').should('not.be.checked')

Content Assertions

cy.get('element').should('contain', 'text')
cy.get('element').should('have.text', 'exact text')
cy.get('element').should('have.value', 'value')
cy.get('element').should('have.attr', 'attribute', 'value')

Alert & Dialog Handling

// Simple Alert
cy.on('window:alert', (str) => {
    expect(str).to.equal('Alert message')
})

// Confirmation Alert - Accept
cy.on('window:confirm', (str) => {
    expect(str).to.equal('Confirm message')
    return true  // Accept
})

// Prompt Alert
cy.window().then(($win) => {
    cy.stub($win, 'prompt').returns('Hello Alert')
    cy.contains('Click for JS Prompt').click()
})

API Testing

// GET Request
cy.request('GET', 'https://api.example.com/users')
  .then((response) => {
      expect(response.status).equal(200)
      expect(response.body.data[0].name).equal('John')
  })

// POST Request
const user = { "name": "John", "job": "developer" }
cy.request('POST', 'https://api.example.com/users', user)
  .then((response) => {
      expect(response.status).equal(201)
      expect(response.body.name).equal(user.name)
  })

Custom Commands

// In cypress/support/commands.js
Cypress.Commands.add("customLogin", (email, password) => {
    cy.visit('/')
    cy.contains('Sign in').click()
    cy.get('input[type="email"]').type(email)
    cy.get('input[type="password"]').type(password)
    cy.get('button[type="submit"]').click()
})

// Usage in tests
cy.customLogin('user@example.com', 'password')

Waits & Network

cy.wait(5000)                           // Wait for milliseconds
cy.intercept('GET', '/api/users').as('getUsers')
cy.wait('@getUsers')                    // Wait for network request
cy.intercept('POST', '/api/login', { fixture: 'user.json' })

Viewport & Screenshots

cy.viewport(1280, 720)                  // Set viewport size
cy.screenshot()                         // Take screenshot
cy.screenshot('my-image')               // Named screenshot
cy.get('element').screenshot()          // Element screenshot

Cookies & Storage

cy.setCookie('name', 'value')           // Set cookie
cy.getCookie('name')                    // Get cookie
cy.clearCookies()                       // Clear all cookies
cy.clearLocalStorage()                  // Clear local storage
cy.clearSessionStorage()                // Clear session storage

Multiple Elements & File Upload

cy.get('li').each(($el, index) => {     // Iterate elements
    cy.wrap($el).click()
})
cy.get('li').first()                    // First element
cy.get('li').last()                     // Last element
cy.get('input[type="file"]').selectFile('path/to/file.pdf')

Running Tests

npm run open

npm run chrome

# Run tests in Electron (headless)
npm run electron
