Selenium WebDriver Reference Guide

Basic Setup

WebDriver Initialization

// Java
WebDriver driver = new ChromeDriver();
WebDriver driver = new FirefoxDriver();
WebDriver driver = new EdgeDriver();

// Python
from selenium import webdriver
driver = webdriver.Chrome()
driver = webdriver.Firefox()
driver = webdriver.Edge()

// C#
IWebDriver driver = new ChromeDriver();
IWebDriver driver = new FirefoxDriver();

Navigation

// Java
driver.get("https://example.com");
driver.navigate().to("https://example.com");
driver.navigate().back();
driver.navigate().forward();
driver.navigate().refresh();

// Python
driver.get("https://example.com")
driver.back()
driver.forward()
driver.refresh()

// C#
driver.Navigate().GoToUrl("https://example.com");
driver.Navigate().Back();
driver.Navigate().Forward();
driver.Navigate().Refresh();

// Java
driver.findElement(By.id("elementId"));
driver.findElement(By.name("elementName"));
driver.findElement(By.className("className"));
driver.findElement(By.tagName("tagName"));
driver.findElement(By.cssSelector("css"));
driver.findElement(By.xpath("//xpath"));
driver.findElement(By.linkText("Link Text"));
driver.findElement(By.partialLinkText("Partial"));

// Python
driver.find_element(By.ID, "elementId")
driver.find_element(By.NAME, "elementName")
driver.find_element(By.CLASS_NAME, "className")
driver.find_element(By.TAG_NAME, "tagName")
driver.find_element(By.CSS_SELECTOR, "css")
driver.find_element(By.XPATH, "//xpath")

// C#
driver.FindElement(By.Id("elementId"));
driver.FindElement(By.Name("elementName"));
driver.FindElement(By.ClassName("className"));
driver.FindElement(By.TagName("tagName"));
driver.FindElement(By.CssSelector("css"));
driver.FindElement(By.XPath("//xpath"));

User Actions

Clicking & Typing

// Java
element.click();
element.sendKeys("text");
element.clear();
element.submit();

// Python
element.click()
element.send_keys("text")
element.clear()
element.submit()

// C#
element.Click();
element.SendKeys("text");
element.Clear();
element.Submit();

Dropdowns

// Java
Select dropdown = new Select(element);
dropdown.selectByVisibleText("Option Text");
dropdown.selectByValue("value");
dropdown.selectByIndex(0);

// Python
from selenium.webdriver.support.ui import Select
dropdown = Select(element)
dropdown.select_by_visible_text("Option Text")
dropdown.select_by_value("value")
dropdown.select_by_index(0)

// C#
SelectElement dropdown = new SelectElement(element);
dropdown.SelectByText("Option Text");
dropdown.SelectByValue("value");
dropdown.SelectByIndex(0);

Waits

// Java - Explicit Wait
WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(10));
wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("element")));
wait.until(ExpectedConditions.elementToBeClickable(By.id("element")));

// Python - Explicit Wait
from selenium.webdriver.support.wait import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
wait = WebDriverWait(driver, 10)
wait.until(EC.visibility_of_element_located((By.ID, "element")))
wait.until(EC.element_to_be_clickable((By.ID, "element")))

// Implicit Wait
driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(10)); // Java
driver.implicitly_wait(10) # Python

Advanced Actions

// Java - Actions Class
Actions actions = new Actions(driver);
actions.moveToElement(element).perform();        // Hover
actions.dragAndDrop(source, target).perform();   // Drag & Drop
actions.contextClick(element).perform();         // Right click
actions.doubleClick(element).perform();          // Double click

// Python - ActionChains
from selenium.webdriver.common.action_chains import ActionChains
actions = ActionChains(driver)
actions.move_to_element(element).perform()
actions.drag_and_drop(source, target).perform()
actions.context_click(element).perform()

// C#
Actions actions = new Actions(driver);
actions.MoveToElement(element).Perform();
actions.DragAndDrop(source, target).Perform();

JavaScript Execution & Multiple Elements

// Java
JavaScriptExecutor js = (JavaScriptExecutor) driver;
js.executeScript("arguments[0].click();", element);
js.executeScript("window.scrollTo(0, document.body.scrollHeight);");
List elements = driver.findElements(By.className("item"));

// Python
driver.execute_script("arguments[0].click();", element)
driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
elements = driver.find_elements(By.CLASS_NAME, "item")

// C#
IJavaScriptExecutor js = (IJavaScriptExecutor)driver;
js.ExecuteScript("arguments[0].click();", element);
var elements = driver.FindElements(By.ClassName("item"));

Frames & Windows

// Java
driver.switchTo().frame("frameName");            // Switch to frame
driver.switchTo().defaultContent();              // Back to main content
driver.switchTo().window(windowHandle);          // Switch window
Set windows = driver.getWindowHandles(); // Get all windows

// Python
driver.switch_to.frame("frameName")
driver.switch_to.default_content()
driver.switch_to.window(window_handle)
windows = driver.window_handles

// C#
driver.SwitchTo().Frame("frameName");
driver.SwitchTo().DefaultContent();
driver.SwitchTo().Window(windowHandle);

Alert Handling

// Java
Alert alert = driver.switchTo().alert();
alert.accept();                              // Accept alert
alert.dismiss();                             // Dismiss alert
String alertText = alert.getText();          // Get alert text
alert.sendKeys("text");                     // Type in prompt

// Python
alert = driver.switch_to.alert
alert.accept()
alert.dismiss()
alert_text = alert.text
alert.send_keys("text")

// C#
IAlert alert = driver.SwitchTo().Alert();
alert.Accept();
alert.Dismiss();
string alertText = alert.Text;
alert.SendKeys("text");

Cookies & Screenshots

// Java
driver.manage().addCookie(new Cookie("name", "value"));
driver.manage().deleteCookieNamed("name");
driver.manage().deleteAllCookies();
File screenshot = ((TakesScreenshot)driver).getScreenshotAs(OutputType.FILE);

// Python
driver.add_cookie({"name": "value"})
driver.delete_cookie("name")
driver.delete_all_cookies()
driver.save_screenshot("screenshot.png")

// C#
driver.Manage().Cookies.AddCookie(new Cookie("name", "value"));
driver.Manage().Cookies.DeleteCookieNamed("name");
driver.Manage().Cookies.DeleteAllCookies();
var screenshot = ((ITakesScreenshot)driver).GetScreenshot();

Browser Management

// Java
driver.manage().window().maximize();
driver.manage().window().setSize(new Dimension(1024, 768));
driver.getTitle();
driver.getCurrentUrl();
driver.getPageSource();
driver.quit();

// Python
driver.maximize_window()
driver.set_window_size(1024, 768)
driver.title
driver.current_url
driver.page_source
driver.quit()

// C#
driver.Manage().Window.Maximize();
driver.Manage().Window.Size = new Size(1024, 768);
driver.Title;
driver.Url;
driver.PageSource;
driver.Quit();
